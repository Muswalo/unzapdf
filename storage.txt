        function formatDate($date)
        {
            date_default_timezone_set('Africa/Lusaka');
            $now = time();
            $timestamp = strtotime($date);
            $diff = $now - $timestamp;
            if ($diff < 0) {
                return 'Date is in the future';
            }
            $intervals = array(
                31536000 => 'year',
                2592000 => 'month',
                86400 => 'day',
                3600 => 'hour',
                60 => 'minute',
                1 => 'second'
            );
            foreach ($intervals as $secs => $str) {
                if ($diff >= $secs) {
                    $num = floor($diff / $secs);
                    return $num . ' ' . $str . ($num > 1 ? 's' : '') . ' ago';
                }
            }
            return 'just now';
        }
